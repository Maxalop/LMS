CREATE DATABASE PMS;

USE PMS;


CREATE TABLE PMS_PRODUCT(
  ProductId INT NOT NULL AUTO_INCREMENT,
  ProductName VARCHAR(30) NOT NULL,
  DepartmentId INT(2),
  PRIMARY KEY (ProductId)  
  );

SELECT * FROM  PMS_PRODUCT;

CREATE TABLE PMS_MANUFACTURING
(ManufactureId INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
ProductId VARCHAR(5) NOT NULL,
UnitId VARCHAR(5) NOT NULL,
Quantity INT(7) NOT NULL,
Availability VARCHAR(3) NOT NULL,
ProductManufactureDate DATE NOT NULL,
ProductExpiryDate DATE NOT NULL
);


SELECT * FROM  PMS_MANUFACTURING;
